102 static inline AL_U32 AlUart_ll_GetBaudRate(AL_REG BaseAddr, AL_U32 InputClockHz)
103 {
104  AL_U16 Divisor;
105  AL_U32 BaudRate;
106  AL_REG32_SET_BIT(BaseAddr + UART__LCR__OFFSET, UART__LCR__DLAB__SHIFT, AL_FUNC_ENABLE);
107  Divisor = (AL_REG32_READ(BaseAddr + UART__IER_DLH__OFFSET) << 8) | (AL_REG32_READ(BaseAddr + UART__RBR__THR__DLL__OFFSET));
108  AL_REG32_SET_BIT(BaseAddr + UART__LCR__OFFSET, UART__LCR__DLAB__SHIFT, AL_FUNC_DISABLE);
109  BaudRate = (AL_U32)((InputClockHz >> 4) / Divisor);
110  return BaudRate;
111 }
