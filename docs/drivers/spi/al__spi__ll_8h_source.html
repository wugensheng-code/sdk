182 static inline AL_REG AlSpi_ll_SetSpiMasterSlave(AL_REG SpiBaseAddr, AL_SPI_Mode SpiMode)
183 {
184  AL_REG TempSpiBaseAddr;
185 
186  if (SPI_MASTER_MODE == SpiMode) {
187  TempSpiBaseAddr = SpiBaseAddr;
188  if (SpiBaseAddr == SPI0_BASE_ADDR) {
189  AL_REG32_SET_BIT(TOP_NS_CFG_CTRL_SPI_ADDR, TOP_NS_CFG_CTRL_SPI_CTRL_SPI0_SHIFT, 0);
190  } else {
191  AL_REG32_SET_BIT(TOP_NS_CFG_CTRL_SPI_ADDR, TOP_NS_CFG_CTRL_SPI_CTRL_SPI1_SHIFT, 0);
192  }
193  } else {
194  TempSpiBaseAddr = SpiBaseAddr + 0x100;
195  if (SpiBaseAddr == SPI0_BASE_ADDR) {
196  AL_REG32_SET_BIT(TOP_NS_CFG_CTRL_SPI_ADDR, TOP_NS_CFG_CTRL_SPI_CTRL_SPI0_SHIFT, 1);
197  } else {
198  AL_REG32_SET_BIT(TOP_NS_CFG_CTRL_SPI_ADDR, TOP_NS_CFG_CTRL_SPI_CTRL_SPI1_SHIFT, 1);
199  }
200  }
201 
202  return TempSpiBaseAddr;
203 }
