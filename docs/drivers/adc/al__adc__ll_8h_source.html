340 static inline AL_VOID AlAdc_ll_WritePlAdcReg(AL_REG AdcBaseAddr, AL_U8 PlRegAddr, AL_U16 Data)
341 {
342  while (AL_REG32_READ(AdcBaseAddr + ADC_MSTS_OFFSET) & ADC_MSTS_R_CMD_FIFO_F_MASK);
343  AL_REG32_WRITE(AdcBaseAddr + ADC_CMD_FIFO_OFFSET, ADC_WRITE_CMD << 24 | PlRegAddr << 16 | Data);
344 }

346 /*
347  * This function is drain Data FIFO.
348  */
349 static inline AL_VOID AlAdc_ll_DrainDataFifo(AL_REG AdcBaseAddr)
350 {
351  AL_U32 Status = AL_REG32_READ(AdcBaseAddr + ADC_MSTS_OFFSET);
352 
353  while(!(Status & ADC_MSTS_R_DATA_FIFO_E_MASK)) {
354  AL_REG32_READ(AdcBaseAddr + ADC_DATA_FIFO_OFFSET);
355  Status = AL_REG32_READ(AdcBaseAddr + ADC_MSTS_OFFSET);
356  }
357 }

359 /*
360  * This function is used for reading to PL-ADC Registers using the Data FIFO.
361  */
362 static inline AL_U16 AlAdc_ll_ReadPlAdcReg(AL_REG AdcBaseAddr, AL_U8 PlRegAddr)
363 {
364  AlAdc_ll_DrainDataFifo(AdcBaseAddr);
365 
366  AL_REG32_WRITE(AdcBaseAddr + ADC_CMD_FIFO_OFFSET, ADC_READ_CMD << 24 | PlRegAddr << 16);
367  while (AL_REG32_READ(AdcBaseAddr + ADC_MSTS_OFFSET) & ADC_MSTS_R_DATA_FIFO_E_MASK);
368  return (AL_U16)(AL_REG32_READ(AdcBaseAddr + ADC_DATA_FIFO_OFFSET));
369 }
